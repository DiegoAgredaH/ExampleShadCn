## Usage

```tsx
import {
  Widget,
  WidgetContent,
  WidgetDescription,
  WidgetFooter,
  WidgetHeader,
  WidgetTitle,
} from "@/components/ui/widget"
```

```tsx
<Widget>
  <WidgetHeader>
    <WidgetTitle>Widget Title</WidgetTitle>
    <WidgetDescription>Widget Description</WidgetDescription>
  </WidgetHeader>
  <WidgetContent>
    <p>Widget Content</p>
  </WidgetContent>
  <WidgetFooter>
    <p>Widget Footer</p>
  </WidgetFooter>
</Widget>
```
